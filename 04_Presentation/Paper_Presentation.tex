%==== 01 - Motivation =========================================================%

\section{Motivation}
\begin{frame}{From Univariate to Multivariate Stochastic Volatility}
  \begin{itemize}
    \setlength\itemsep{0.9em}

    \item \textbf{Recall Assignment Task 2:} We modelled daily CHF/USD returns
    with $t_\nu$-errors and latent weights $\omega_i$, giving each observation
    its own effective variance $\sigma^2/\omega_i$ --- a \emph{univariate}
    stochastic volatility (SV) model.

    \item \textbf{Natural extension:} With $m$ assets we need not only
    time-varying \emph{variances} but also time-varying \emph{covariances}.
    The full covariance matrix has $\tfrac{m(m+1)}{2}$ free elements ---
    growing \textbf{quadratically} in $m$.

    \item \textbf{The curse of dimensionality:} For $m=26$ exchange rates
    this gives 351 covariance parameters \emph{per time point}.
    Direct modelling is infeasible.

    \item \textbf{This paper's answer:} Impose a low-dimensional
    \textbf{factor structure} ($r \ll m$ latent factors) and let \emph{each}
    factor follow its own SV process --- parsimonious yet flexible.

  \end{itemize}
\end{frame}

%==== 02 - The Model ==========================================================%

\section{The Factor SV Model}

\begin{frame}{Building Block 1: The Factor SV Model}

  \textbf{Intuition:} Decompose $m$-dimensional returns into a
  low-dimensional common component (factors) plus series-specific noise.
  Both components have \emph{time-varying} volatility.

  \vspace{0.4em}

  \begin{alertblock}{Observation \& Factor Equations}
    \[
      \underbrace{\mathbf{y}_t}_{m \times 1}
      = \underbrace{\boldsymbol{\Lambda}}_{m \times r}
        \underbrace{\mathbf{f}_t}_{r \times 1}
      + \underbrace{\boldsymbol{\varepsilon}_t}_{m \times 1},
      \qquad
      \mathbf{f}_t \mid \mathbf{h}_t \sim \mathcal{N}_r(\mathbf{0}, V_t),
      \quad
      \boldsymbol{\varepsilon}_t \mid \mathbf{h}_t \sim \mathcal{N}_m(\mathbf{0}, U_t)
    \]
  \end{alertblock}

  \begin{itemize}
    \setlength\itemsep{0.45em}
    \item $\boldsymbol{\Lambda}$: $m \times r$ \textbf{factor loadings} ---
          how much each asset responds to each factor
    \item $U_t = \mathrm{diag}(e^{h_{1t}}, \ldots, e^{h_{mt}})$:
          \textbf{idiosyncratic} time-varying variances
    \item $V_t = \mathrm{diag}(e^{h_{m+1,t}}, \ldots, e^{h_{m+r,t}})$:
          \textbf{factor} time-varying variances
    \item Each log-variance follows an independent \textbf{AR(1)}:
          $h_{it} = \mu_i + \phi_i(h_{i,t-1}-\mu_i) + \sigma_i\eta_{it}$
  \end{itemize}

  \vspace{0.2em}

  \begin{block}{Key quantity: time-varying covariance matrix}
    \[
      \mathrm{cov}(\mathbf{y}_t \mid \mathbf{h}_t)
      \;=\; \boldsymbol{\Lambda} V_t \boldsymbol{\Lambda}^\prime + U_t
    \]
    % \textit{All co-movements are governed by the $r$ factors.
    % Idiosyncratic shocks are uncorrelated across assets.}
  \end{block}

\end{frame}

%==== 03 - Identification =====================================================%

\begin{frame}{Building Block 2: Identification}

  \textbf{Problem:} The factor model has a fund. \textbf{scaling ambiguity}.
  For any scalar $c \neq 0$:
  \[
    \boldsymbol{\Lambda}\mathbf{f}_t
    \;=\; \underbrace{(c\,\boldsymbol{\Lambda})}_{\boldsymbol{\Lambda}^*}
          \underbrace{(c^{-1}\mathbf{f}_t)}_{\mathbf{f}_t^*}
  \]
  The data cannot distinguish $(\boldsymbol{\Lambda}, \mathbf{f}_t)$ from
  $(c\boldsymbol{\Lambda}, c^{-1}\mathbf{f}_t)$ --- the model is
  \textbf{not identified} without additional constraints.

  \vspace{0.05em}

  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Classical fix}
        Set $\lambda_{jj} = 1$ for each factor,
        treat the factor variance level $\mu_{m+j}$ as free.
        \begin{itemize}
          \item[$+$] Simple to implement
          \item[$-$] Imposes a series ordering
          \item[$-$] Creates artificial ``leading'' factors
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{alertblock}{Paper's approach}
        Fix $\mu_{m+j} = 0$ (factor variance averages
        to 1 over time), leave $\lambda_{jj}$
        \emph{unrestricted}.
        \begin{itemize}
          \item[$+$] No ordering imposed
          \item[$+$] Leadership shared across series
          \item[$+$] Residual ambiguity in $\lambda_{jj}$
                becomes a \textbf{computational tool}
        \end{itemize}
      \end{alertblock}
    \end{column}
  \end{columns}

  % \vspace{0.1em}

  \begin{block}{Why this matters for MCMC}
    Unresolved non-identifiability $\Rightarrow$ multimodal posterior
    $\Rightarrow$ MCMC chains get stuck.
  \end{block}

\end{frame}

%==== 04 - Priors & Data Augmentation =========================================%

\section{Bayesian Inference}

\begin{frame}{Building Block 3: Priors \& Data Augmentation}

  \textbf{Priors} (independently for each SV process $i = 1,\ldots,m+r$):
  \[
    \mu_i \sim \mathcal{N}(b_\mu, B_\mu), \qquad
    \frac{\phi_i+1}{2} \sim \mathcal{B}(a_0,b_0), \qquad
    \sigma_i^2 \sim B_\sigma \cdot \chi^2_1
  \]
  For each unrestricted element of $\boldsymbol{\Lambda}$:
  $\lambda_{ij} \sim \mathcal{N}(0, B_\lambda)$
  \hfill \textit{(Ridge-type shrinkage of small loadings)}

  \vspace{0.5em}

  \textbf{Data augmentation:} Introduce latent variables
  $\mathbf{h} = \{h_{i,\bullet}\}$ (log-volatility paths) and
  $\mathbf{f} = \{\mathbf{f}_{t}\}$ (factor paths) as ``missing data''.

  \vspace{0.3em}

  \begin{alertblock}{Why augmentation helps --- the key idea}
    \begin{itemize}
      \setlength\itemsep{0.4em}
      \item \textbf{Conditional on $\mathbf{f}$ and $\boldsymbol{\Lambda}$:}
            each of the $m+r$ series reduces to an independent
            \textbf{univariate SV model} $\Rightarrow$ use \texttt{stochvol}.
      \item \textbf{Conditional on $\mathbf{h}$:}
            $\boldsymbol{\Lambda}$ and $\mathbf{f}_t$ have
            \textbf{closed-form Normal posteriors}
            $\Rightarrow$ standard regression updates.
    \end{itemize}
    \vspace{0.2em}
    \textit{Direct parallel to Task~2: augmenting with $\omega_i$ turned an
    intractable $t$-likelihood into a conjugate Normal--Gamma system.}
  \end{alertblock}

\end{frame}

%==== 05 - Standard Gibbs =====================================================%

\begin{frame}{Building Block 4: The Standard Gibbs Sampler}

  Iterate the following three steps \citep[Algorithm~1]{Kastner2017}:

  \vspace{0.3em}

  \begin{alertblock}{Step (a) --- Univariate SV updates ($m+r$ independent problems)}
    Conditional on $\mathbf{f}$ and $\boldsymbol{\Lambda}$, use the
    log-squared observation trick to write each series as a linear
    state-space model and update $(h_{i,\bullet}, \mu_i, \phi_i, \sigma_i)$
    via \texttt{stochvol}.
  \end{alertblock}

  \begin{alertblock}{Step (b) --- Sample loadings $\boldsymbol{\Lambda}$}
    Conditional on $\mathbf{f}$ and $\mathbf{h}$, each row of
    $\boldsymbol{\Lambda}$ is a \textbf{weighted Normal regression} posterior
    --- $m$ independent problems, each with $T$ observations,
    weighted by $e^{-h_{it}}$.
  \end{alertblock}

  \begin{alertblock}{Step (c) --- Sample factors $\mathbf{f}_t$}
    Conditional on $\boldsymbol{\Lambda}$ and $\mathbf{h}$, each
    $\mathbf{f}_t$ is a \textbf{multivariate Normal} posterior ---
    $T$ independent $r$-dimensional updates.
  \end{alertblock}

  \vspace{0.3em}

  \textbf{In principle correct --- in practice broken.}  See next slide.

\end{frame}

%==== 06 - Why Gibbs Fails ====================================================%

\begin{frame}{The Problem: Standard Gibbs Mixes Poorly}

  \begin{itemize}
    \setlength\itemsep{0.7em}

    \item \textbf{Root cause:} $\boldsymbol{\Lambda}$ and $\mathbf{f}_t$
    are \textbf{strongly correlated} in the joint posterior.
    If $\lambda_{jj}$ is large, the factors $f_{jt}$ must be small to keep
    $\Lambda f_t$ constant, and vice versa.
    Sequential updates cannot break this ridge-shaped dependence.

    \item \textbf{Consequence:} The chain inch-worms along a narrow ridge
    rather than freely exploring the parameter space.

    \item \textbf{Simulation evidence} ($m=10$, $r=2$, $T=1000$, 100 datasets):
    \textbf{IF for $\lambda_{11}$ exceeds 5000} on some datasets even after
    5 million iterations --- an effective sample size of only $\sim$1000
    from 5 million draws.

  \end{itemize}

  \vspace{0.2em}

  \begin{block}{Inefficiency Factor (IF) --- reminder}
    $\mathrm{IF} = k$ means you need $k$ correlated MCMC draws to match the
    accuracy of \emph{one} independent draw.
    \\[0.3em]
    \textit{IF = 5000 with $M=100{,}000$ draws $\Rightarrow$
    effective sample size of just 20.  Practically useless for inference.}
  \end{block}

\end{frame}

%==== 07 - ASIS Key Idea ======================================================%

\section{The ASIS Innovation}
\label{sec:asis}

\begin{frame}{The ASIS Key Idea}

  \textbf{Core insight \citep{Yu2011}:} Two \emph{mathematically equivalent}
  parameterisations are often each efficient in \emph{different} regions of
  the param. space.  \textbf{Switching between them at every iteration}
  combines the best of both worlds.

  \vspace{0.1em}

  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Baseline param.}
        \[
          \mathbf{y}_t = \boldsymbol{\Lambda}\mathbf{f}_t
          + \boldsymbol{\varepsilon}_t
        \]
        $\lambda_{jj}$ \textbf{free}, $\mu_{m+j}=0$ fixed\\[0.3em]
        Factor variance fixed at scale 1 on average.
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Alternative param.}
        \[
          \mathbf{y}_t = \boldsymbol{\Lambda}^*\mathbf{f}_t^*
          + \boldsymbol{\varepsilon}_t
        \]
        $\lambda^*_{jj}=1$ \textbf{fixed}, $\mu_{m+j}$ free\\[0.3em]
        Scaling absorbed into the SV level.
      \end{block}
    \end{column}
  \end{columns}

  \vspace{0.1em}

  \textbf{The link} (a simple change of variables):
  \[
    \mathbf{f}_t^* = \lambda_{jj}\,\mathbf{f}_t,
    \qquad
    \boldsymbol{\Lambda}^* = \lambda_{jj}^{-1}\boldsymbol{\Lambda}
  \]

  \vspace{0.1em}

  \begin{alertblock}{ASIS in one sentence}
    After each standard Gibbs step, transform to the alternative
    parameterisation, \textbf{resample $\lambda_{jj}$} there (where the
    posterior is better-behaved), then transform back.\\
    No extra model fitting --- just a change of coordinates at each step.
  \end{alertblock}

\end{frame}

%==== 08 - Shallow & Deep =====================================================%

\begin{frame}{Shallow vs.\ Deep Interweaving}

  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{alertblock}{Shallow interweaving}
        \textbf{Transform:}
        $\mathbf{f}_t^* = \lambda_{jj}\mathbf{f}_t$

        \vspace{0.1em}
        \textbf{Resample} $\lambda_{jj}^2$ from a conjugate
        \textbf{GIG} posterior (see Appendix~A1):
        \[
          \lambda_{jj}^2 \mid \cdot \;\sim\;
          \mathrm{GIG}(p,\,a,\,b)
        \]

        \vspace{0.1em}
        \textbf{Transform back:}
        $\mathbf{f}_t = (\lambda_{jj}^{\mathrm{new}})^{-1}\mathbf{f}_t^*$

        \vspace{0.1em}
        \textbf{Scope:} $\boldsymbol{\Lambda}$ and $\mathbf{f}$ only.

        \vspace{0.3em}
        \textbf{Gain:} $\approx 2$--$8\times$
      \end{alertblock}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{alertblock}{Deep interweaving}
        \textbf{Transform:} additionally shift the
        log-volatility path (see Appendix~A2):
        \[
          h^*_{m+j,t} = h_{m+j,t} + 2\log|\lambda_{jj}|
        \]
        Now $\mu_{m+j} = \log\lambda_{jj}^2$ is free.

        \vspace{0.3em}
        \textbf{Resample} $\lambda_{jj}$ via
        \textbf{independence MH} using a Gaussian
        approximation to the conditional likelihood.

        \vspace{0.3em}
        \textbf{Scope:} $\boldsymbol{\Lambda}$, $\mathbf{f}$
        \emph{and} $\mathbf{h}$.

        \vspace{0.3em}
        \textbf{Gain:} up to $\mathbf{400\times}$
      \end{alertblock}
    \end{column}
  \end{columns}

  \vspace{0.2em}

  \begin{block}{Why deep beats shallow}
    Shallow leaves the log-volatility path $h_{m+j,\bullet}$ frozen,
    so the slow mixing between $\lambda_{jj}$ and $h_{m+j,\bullet}$
    persists.  Deep interweaving breaks \emph{both} dependencies
    simultaneously.  \textbf{Neither strategy requires tuning.}
  \end{block}

\end{frame}

%==== A1 - GIG Derivation =====================================================%

\begin{frame}{A1: Why the GIG Posterior in Shallow Interweaving}

  \textbf{Goal:} derive $p(\lambda_{jj}^2 \mid \mathbf{f}^*, \boldsymbol{\Lambda}^*, \mathbf{h})$.
  Let $\psi = \lambda_{jj}^2$.  Three components contribute:

  \vspace{0.3em}

  \textbf{(1) Likelihood from the scaled factors}
  $f_{jt}^* \mid h_{m+j,t}, \psi \sim \mathcal{N}(0, \psi\, e^{h_{m+j,t}})$:
  \[
    p(\mathbf{f}_j^* \mid \mathbf{h}, \psi)
    \;\propto\; \psi^{-T/2}
    \exp\!\left(-\frac{1}{2\psi}\sum_t \frac{(f_{jt}^*)^2}{e^{h_{m+j,t}}}\right)
  \]

  \textbf{(2) Induced prior on transformed loadings}
  $\lambda^*_{ij} \mid \psi \sim \mathcal{N}(0, B_\lambda/\psi)$:
  \[
    p(\boldsymbol{\lambda}^*_{-jj} \mid \psi)
    \;\propto\; \psi^{k_j/2}
    \exp\!\left(-\frac{\psi}{2B_\lambda}\|\boldsymbol{\lambda}^*_{-jj}\|^2\right)
  \]

  \textbf{(3) Prior} $\psi \sim \mathcal{G}(1/2,\, 1/(2B_\lambda))$:
  $\;\propto\; \psi^{-1/2} e^{-\psi/(2B_\lambda)}$

  \vspace{0.3em}

  \textbf{Combining all three} gives a kernel of the form
  $\psi^{p-1}\exp\{-(a\psi + b/\psi)/2\}$, which is the
  \textbf{Generalised Inverse Gaussian} $\mathrm{GIG}(p,a,b)$ with:
  \[
    p = \tfrac{1+k_j-T}{2}, \quad
    a = \tfrac{1}{B_\lambda}\bigl(1+\|\boldsymbol{\lambda}^*_{-jj}\|^2\bigr), \quad
    b = \sum_t \tfrac{(f_{jt}^*)^2}{e^{h_{m+j,t}}}
  \]
  Sampled efficiently via the \texttt{GIGrvg} R package.

\end{frame}

%==== A2 - Deep Interweaving Derivation =======================================%

\begin{frame}{A2: The Log-Volatility Shift in Deep Interweaving}

  \textbf{Starting point:} In the baseline param., $\mathrm{Var}(f_{jt}) = e^{h_{m+j,t}}$
  with $\mu_{m+j}=0$ fixed.

  \vspace{0.3em}

  \textbf{Step 1 --- absorb $\lambda_{jj}$ into the SV level.}
  The scaled factor $f_{jt}^* = \lambda_{jj} f_{jt}$ has variance:
  \[
    \mathrm{Var}(f_{jt}^*)
    = \lambda_{jj}^2\, e^{h_{m+j,t}}
    = \exp\!\bigl(\underbrace{\log\lambda_{jj}^2 + h_{m+j,t}}_{=:\, h^*_{m+j,t}}\bigr)
  \]
  So the transformed log-variance is $h^*_{m+j,t} = h_{m+j,t} + 2\log|\lambda_{jj}|$,
  and the level of the new AR(1) process is $\mu_{m+j} = \log\lambda_{jj}^2$.

  \vspace{0.3em}

  \textbf{Step 2 --- resample the level.}
  The conditional likelihood
  $p(h^*_{m+j,1},\ldots,h^*_{m+j,T} \mid \mu_{m+j}, \phi, \sigma^2)$
  is Gaussian in $\mu_{m+j}$, so a tight \textbf{independence MH} proposal
  is constructed analytically (matching mean and variance of this likelihood).

  \vspace{0.3em}

  \textbf{Step 3 --- transform back.}
  Given $\mu_{m+j}^{\mathrm{new}}$, set
  $\lambda_{jj}^{\mathrm{new}} = e^{\mu_{m+j}^{\mathrm{new}}/2}$ and recover:
  \[
    h_{m+j,t}^{\mathrm{new}}
    = h^*_{m+j,t} - 2\log|\lambda_{jj}^{\mathrm{new}}|,
    \qquad
    \boldsymbol{\Lambda}^{\mathrm{new}} = \frac{\lambda_{jj}^{\mathrm{new}}}{\lambda_{jj}^{\mathrm{old}}}\boldsymbol{\Lambda}^{\mathrm{old}}
  \]

\end{frame}

\begin{frame}{A3: Connection to Thesis --- Crash-Filtered Equity Index}

  \textbf{Thesis goal \citep{Tewari2024}:} Predict \textbf{Catastrophic Stock
  Implosion (CSI)} events ($>80\%$ drawdown + zombie period) and construct a
  crash-filtered index with superior risk-adjusted returns.

  \vspace{0.3em}

  \textbf{Where the factor SV model fits:}

  \begin{itemize}
    \setlength\itemsep{0.6em}

    \item \textbf{Feature engineering for CSI prediction:}
    The idiosyncratic log-variance $h_{it}$ and factor log-variances
    $h_{m+j,t}$ are natural inputs to the ensemble classifiers (RF, XGBoost).
    A firm whose \emph{idiosyncratic} SV is elevated and detaches from the
    common factors is a candidate for implosion --- this is precisely what
    $U_t$ captures, separately from $\Lambda V_t \Lambda^\prime$.

    \item \textbf{Dynamic covariance in the optimiser:}
    After filtering CSI candidates, the posterior mean of $\Sigma_t$
    replaces the sample covariance in the Markowitz step at each annual
    rebalancing --- more stable and current than backward-looking estimates.

    \item \textbf{Uncertainty-aware threshold:}
    The posterior distribution of $\Sigma_t$ can propagate estimation
    uncertainty into the FPR constraint, making the exclusion threshold
    $\theta$ \emph{data-adaptive} rather than fixed.

  \end{itemize}

\end{frame}
